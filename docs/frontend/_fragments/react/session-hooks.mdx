Session lifecycle hooks (`useSessionStart`, `useSessionVerify`, `useSessionRefresh`, etc.) are **not available** with a publishable key. Start sessions on your backend with `@ductape/sdk`, then use the token in frontend hooks.

### Auth context pattern

```tsx
import { createContext, useContext, useState, useCallback } from 'react';
import { DuctapeProvider } from '@ductape/react';

type AuthContextType = {
  sessionToken: string | null;
  login: (email: string, password: string) => Promise<void>;
  logout: () => Promise<void>;
};

const AuthContext = createContext<AuthContextType | null>(null);

export function AuthProvider({ children }: { children: React.ReactNode }) {
  const [sessionToken, setSessionToken] = useState<string | null>(
    () => localStorage.getItem('ductape_session'),
  );

  const login = useCallback(async (email: string, password: string) => {
    const res = await fetch('/api/auth/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email, password }),
    });
    const { sessionToken: token, refreshToken } = await res.json();
    localStorage.setItem('ductape_session', token);
    localStorage.setItem('ductape_refresh', refreshToken);
    setSessionToken(token);
  }, []);

  const logout = useCallback(async () => {
    await fetch('/api/auth/logout', { method: 'POST' });
    localStorage.removeItem('ductape_session');
    localStorage.removeItem('ductape_refresh');
    setSessionToken(null);
  }, []);

  return (
    <AuthContext.Provider value={{ sessionToken, login, logout }}>
      <DuctapeProvider
        config={{
          publishableKey: import.meta.env.VITE_DUCTAPE_PK,
          product: 'my-product',
        }}
      >
        {children}
      </DuctapeProvider>
    </AuthContext.Provider>
  );
}

export function useAuth() {
  const ctx = useContext(AuthContext);
  if (!ctx) throw new Error('useAuth requires AuthProvider');
  return ctx;
}
```

### Pass `session` in hooks

```tsx
import { useDatabaseQuery } from '@ductape/react';
import { useAuth } from './AuthProvider';

function UsersList() {
  const { sessionToken } = useAuth();

  const { data, isLoading, error } = useDatabaseQuery('users', {
    table: 'users',
    limit: 20,
    session: sessionToken!, // required with publishable key
  }, { enabled: !!sessionToken });

  if (!sessionToken) return <p>Please log in.</p>;
  if (isLoading) return <p>Loading…</p>;
  if (error) return <p>Error: {error.message}</p>;

  return (
    <ul>
      {data?.rows?.map((row) => (
        <li key={row.id}>{row.email}</li>
      ))}
    </ul>
  );
}
```

The same `session` option applies to `useStorageUpload`, `useActionRun`, feature hooks, and other Ductape React hooks.
